剑指offer 66题个人题解

1. 二维数组查找

在一个二维数组中（每个一维数组的长度相同），每一行都按照从左到右递增的顺序排序，每一列都按照从上到下递增的顺序排序。请完成一个函数，输入这样的一个二维数组和一个整数，判断数组中是否含有该整数。leetcode 74
方法一：暴力枚举 row major order 遍历 O(mn)效率很差
方法二：从右上角开始step search每一步比当前元素小就往左走 比当前元素大就往下走
step search on sorted matrix worst case O(m+n) 需要行内严格排序 行与行之间可以不排序

特殊情况：若矩阵严格row major order排序，则可以使用binary search O(lg(m+n))
例如

[10, 11, 16, 20]
[23, 30, 34, 50]
binary search on 2d fully sorted row major order matrix

2. 替换空格

请实现一个函数，将一个字符串中的每个空格替换成“%20”。例如，当字符串为We Are Happy.则经过替换之后的字符串为We%20Are%20Happy
方法一：直接遍历替换，每一次遇到空格替换为%20并将后续字符串后移2位 O(N^2)
方法二：O(N) TIME O(N) SPACE WORST CASE (All space多分配两倍内存)

3. 从尾到头打印链表

输入一个链表，按链表值从尾到头的顺序返回一个ArrayList。 leetcode 206
方法一：反转链表之后把值遍历加进arraylist O(N) TIME O(1) SPACE（如果可以inplace modify linkedlist）

方法二：使用stack来反转值的顺序 因为stack是顺序储存先进先出 可以起到反转顺序的作用 O(N) TIME O(N) SPACE

4. 重建二叉树

输入某二叉树的前序遍历和中序遍历的结果，请重建出该二叉树。假设输入的前序遍历和中序遍历的结果中都不含重复的数字。例如输入前序遍历序列{1,2,4,7,3,5,6,8}和中序遍历序列{4,7,2,1,5,3,8,6}，则重建二叉树并返回。

首先我们知道preorder[0]是root 通过找到他在inorder中的位置我们可以知道左子树右子树的节点数量和具体节点
对于每一个子树的节点列表 preorder’’[0]仍是子树的根这样递归的进行树的重建。 但是对于非平衡树 或者极端情况一条链表 在inorder里搜索跟的位置要时间复杂度O(n^2) 并且取决于怎样split我们甚至可能需要空间复杂度O(n^2) 我们可以使用map来降低时间空间复杂度 但仍要O(n)
若不使用map我们也可以达到时间O(n)空间O(n)我们手动设置递归终止条件：
每个stack frame上我们使用当前子树根节点的值作为其左子树的递归终止条件 右子树使用当前的stop值作为递归终止条件
Instead of finding a subtree root in inorder, splitting the arrays into parts and recursing on them, just recurse on the full remaining arrays and stop when you come across the subtree root in inorder. Each recursive call gets told where to stop, and it tells its subcalls where to stop. It gives its own root value as stopper to its left subcall and its parent‘s stopper as stopper to its right subcall.
preorder = [3,9,20,15,7]
inorder = [9,3,15,20,7]
递归树：

class Solution {
private int in = 0, pre = 0;
public TreeNode buildTree(int[] preorder, int[] inorder) {
return buildTree(preorder, inorder, Integer.MIN_VALUE);
}
public TreeNode buildTree(int[] preorder, int[] inorder, int stop) {
if(pre >= preorder.length) return null;
if(inorder[in] == stop) {in++;return null;}
TreeNode subtree_root = new TreeNode(preorder[pre++]);//establish current subtree root
subtree_root.left=buildTree(preorder,inorder,subtree_root.val);
subtree_root.right=buildTree(preorder,inorder,stop);
return subtree_root;
}
}

class Solution {
private int in = 0, post = 0;
public TreeNode buildTree(int[] inorder, int[] postorder) {
if(inorder.length!=postorder.length || inorder.length == 0) return null;
int[] reversed_postorder = new int[postorder.length];
int[] reversed_inorder = new int[inorder.length];
for(int i = 0; i < postorder.length;i++) {
reversed_postorder[i]=postorder[postorder.length-1-i];
reversed_inorder[i]=inorder[inorder.length-1-i];
}
return buildTree(reversed_inorder, reversed_postorder,Integer.MIN_VALUE);
}
public TreeNode buildTree(int[] inorder, int[] postorder, int stop) {
if(post >= postorder.length) return null;
if(inorder[in] == stop) {in++;return null;}
TreeNode subtree_root = new TreeNode(postorder[post++]);//establish current subtree root
subtree_root.right=buildTree(inorder,postorder,subtree_root.val);
subtree_root.left=buildTree(inorder,postorder,stop);
return subtree_root;
}
}

output = new Stack();
}

/** Push element x to the back of queue. */
public void push(int x) {
    input.push(x);
}

/** Removes the element from in front of queue and returns that element. */
public int pop() {
    peek();
    return output.pop(); 
}

/** Get the front element. */
public int peek() {
    if (output.empty())
        while (!input.empty())
            output.push(input.pop());
    return output.peek();
}

/** Returns whether the queue is empty. */
public boolean empty() {
    return input.isEmpty() && output.isEmpty();
}

}

class Solution {
public int findMin(int[] nums) {
if(numsnull||nums.length0) return -1;//boundary case!
int l = 0, r = nums.length - 1, m = 0;
while(l<=r){
m = (l+r)/2;
if(nums[m] >= nums[r]) l = m+1;
else r = m;
}
return nums[m];
}
}

private Integer[] memo;

public int fib(int N) {
memo = new Integer[N + 1];
return helper(N);
}

private int helper(int N) {
if (memo[N] != null) return memo[N];
if (N <= 1) {
memo[N] = N;
return N;
}
memo[N] = helper(N - 1) + helper(N - 2);
return memo[N];
}

fib[0]=0;fib[1]=1;

return fib[N];
}

public class Solution {
public int JumpFloorII(int n) {
if(n >= 0 && n <= 1) return n;
int f[] = new int[n + 1];
f[0] = 0; f[1] = 1;
for (int i = 2; i < n + 1; i++) {
f[i] = 2 * f[i-1];
}
return f[n];
}
}

class Solution {
public int[] maxSlidingWindow(int[] nums, int k) {
if(nums.length==0 || k<=0) return new int[0];
int[] result =new int[nums.length-k+1];
for(int i = 0; i < nums.length-k+1; i++){
int max=Integer.MIN_VALUE;
for(int j = i;j<i+k;j++) max = max>=nums[j]?max:nums[j];
result[i]=max;
}
return result;
}
}

class Solution {
public int[] maxSlidingWindow(int[] nums, int k) {
if(nums.length==0 || k <=0) return new int[0];
int[] result = new int[nums.length - k + 1];
Deque dq = new ArrayDeque<>();
for(int i = 0,j=0; i<nums.length;i++) {
while(!dq.isEmpty() && dq.peek()<i-k+1) dq.poll();
while(!dq.isEmpty() && nums[dq.peekLast()]<nums[i]) dq.pollLast();
dq.offer(i);
if(i >= k-1) result[j++] = nums[dq.peek()];
}
return result;
}
}
